153  subroutine read_from_file_ctmp2d(mp, name_root)

155  character(len=*), intent(in) :: name_root
156  character(len=256) :: label
157  character(len=256) :: filename_local
158  character(len=256) :: filename
159  sll_int32 :: io_stat
160  sll_int32 :: input_file_id
161  sll_int32 :: i
162  sll_int32 :: ierr
163  sll_int32 :: whatisthisrd
164  sll_int32 :: dimension
165  sll_int32 :: number_patches
166  sll_int32, dimension(:), pointer :: connectivities
167  sll_int32, dimension(:, :), pointer :: connectivities_reshaped
168  character(len=128), dimension(:), pointer :: patches
169 
170  ! local variables for element connectivity
171  sll_int32, dimension(:), allocatable :: global_indices_array
172  sll_int32, dimension(:), allocatable :: local_spline_indices_array
173  sll_int32, dimension(:), allocatable :: local_to_global_index_array
174  sll_int32 :: num_global_indices
175  sll_int32 :: rows
176  sll_int32 :: cols
177  sll_int32 :: li_total
178  sll_int32 :: l2g_total
179  character(len=128), dimension(:), allocatable :: local_index_file_list
180  character(len=128), dimension(:), allocatable :: local_to_global_file_list
181 
182  namelist /multipatch_label/ label
183  namelist /rd/ whatisthisrd
184  namelist /dim/ dimension
185  namelist /num_patches/ number_patches
186  namelist /connectivity/ connectivities
187  namelist /patch_names/ patches
188  namelist /number_global_indices/ num_global_indices
189  namelist /global_indices/ global_indices_array
190  namelist /local_index_files/ local_index_file_list
191  namelist /local_to_global_files/ local_to_global_file_list
192  namelist /li_dimensions/ rows, cols, li_total
193  namelist /l2g_dimensions/ rows, cols, l2g_total
194  namelist /local_spline_indices/ local_spline_indices_array
195  namelist /local_to_global_indices/ local_to_global_index_array
196 
197  if (len(name_root) >= 128) then
198  print *, 'ERROR, read_from_file_ctmp2d => ', &
199  'root names longer than 128 characters are not allowed.'
200  stop
201  end if
202 
203  mp%name_root = trim(name_root)
204 
205  ! We explore the idea that a multipatch geometry is described by multiple
206  ! files, all sharing a 'root name'. By passing the root name, each
207  ! subroutine is then able to build the proper filename to open.
208  filename_local = trim(mp%name_root)//"_info.nml"
209 
210  ! get a new identifier for the file.
211  call sll_s_new_file_id(input_file_id, ierr)
212  if (ierr .ne. 0) then
213  print *, 'ERROR while trying to obtain an unique identifier for file ', &
214  filename_local, '. Called from read_from_file_ctmp2d().'
215  stop
216  end if
217  open (unit=input_file_id, file=filename_local, status="OLD", iostat=io_stat)
218  if (io_stat .ne. 0) then
219  print *, 'ERROR while opening file ', filename_local, &
220  '. Called from read_from_file_ctmp2d().'
221  stop
222  end if
223 
224  ! read the label
225  read (input_file_id, multipatch_label)
226  ! read the rd
227  read (input_file_id, rd)
228  print *, 'rd = ', whatisthisrd
229  read (input_file_id, dim)
230  print *, 'dimension =', dimension
231  read (input_file_id, num_patches)
232  mp%number_patches = number_patches
233  print *, 'number patches = ', number_patches
234  sll_allocate(connectivities(number_patches*8), ierr)
235  sll_allocate(connectivities_reshaped(8, number_patches), ierr)
236  sll_allocate(mp%transfs(number_patches), ierr)
237 
238  read (input_file_id, connectivity)
239  connectivities_reshaped = reshape(connectivities, (/8, number_patches/))
240  print *, 'connectivity array: ', connectivities_reshaped
241  sll_allocate(mp%connectivities(number_patches, 8), ierr)
242  mp%connectivities = transpose(connectivities_reshaped)
243  print *, 'size mp%connectivities = ', size(mp%connectivities, 1), &
244  size(mp%connectivities, 2), mp%connectivities
245  sll_allocate(patches(number_patches), ierr)
246  read (input_file_id, patch_names)
247  print *, 'patch names: ', patches
248  close (input_file_id)
249 
250  ! Create the coordinate transformations.
251  do i = 1, number_patches
252  !SLL_ALLOCATE(mp%transfs(i)%T, ierr)
253  call mp%transfs(i)%read_from_file(trim(patches(i)))
254  end do
255 
256  ! Obtain the information regarding the local index and local to global
257  ! index information for the collection of patches. This is used for finite
258  ! element calculations. This initialization consists of
259  ! several steps:
260  !
261  ! 1. Find out which file to read to get the information on:
262  ! 1.1 global indexing of the elements
263  ! 1.2 the files which contain the local indexing information per patch.
264  ! 1.3 the files which contain the local to global indexing information
265  ! by patch.
266  ! 2. Allocate the arrays which will store the local_index and
267  ! sll_o_local_to_global index information inside the multipatch.
268  ! 3. Loop over the filenames found in step 1. to initialize the arrays.
269 
270  ! The following line establishes a hardwired convetion about the naming
271  ! of the diverse files that represent the multipatch. Not good but works.
272 
273  filename = trim(mp%name_root)//"_element_connectivity_main.nml"
274 
275  sll_allocate(mp%local_indices(number_patches), ierr)
276  sll_allocate(mp%local_to_global_ind(number_patches), ierr)
277 
278  call sll_s_new_file_id(input_file_id, ierr)
279  if (ierr .ne. 0) then
280  print *, 'ERROR while trying to obtain an unique identifier for file ', &
281  trim(filename), '. Called from read_from_file_ctmp2d().'
282  stop
283  end if
284  open (unit=input_file_id, file=trim(filename), status="OLD", iostat=io_stat)
285  if (io_stat .ne. 0) then
286  print *, 'ERROR while opening file ', trim(filename), &
287  '. Called from read_from_file_ctmp2d().'
288  stop
289  end if
290  sll_allocate(local_index_file_list(number_patches), ierr)
291  sll_allocate(local_to_global_file_list(number_patches), ierr)
292 
293  read (input_file_id, number_global_indices)
294  sll_allocate(mp%global_indices(num_global_indices), ierr)
295  sll_allocate(global_indices_array(num_global_indices), ierr)
296 
297  read (input_file_id, global_indices)
298  mp%global_indices(:) = global_indices_array(:)
299  sll_deallocate_array(global_indices_array, ierr)
300  read (input_file_id, local_index_files)
301  read (input_file_id, local_to_global_files)
302 
303  close (input_file_id)
304 
305  ! Read the individual files containing the local index and local to
306  ! global index information.
307  print *, 'initialize_scalar_field_sfmp2d(): reading local index data. '
308 
309  do i = 1, number_patches
310  call sll_s_new_file_id(input_file_id, ierr)
311  if (ierr .ne. 0) then
312  print *, 'ERROR while trying to obtain an unique identifier ', &
313  'for file ', trim(local_index_file_list(i)), &
314  '. Called from read_from_file_ctmp2d().'
315  stop
316  end if
317  open (unit=input_file_id, file=trim(local_index_file_list(i)), &
318  status="OLD", iostat=io_stat)
319  if (io_stat .ne. 0) then
320  print *, 'ERROR while opening file ', &
321  trim(local_index_file_list(i)), &
322  '. Called from read_from_file_ctmp2d().'
323  stop
324  end if
325  read (input_file_id, li_dimensions)
326 
327  print *, 'read_from_file_ctmp2d(): reading data from file: ', &
328  trim(local_index_file_list(i))
329 
330  ! For now we allocate only a linear array, but a 2D array may be
331  ! more useful, especially since we may need to reshape & transpose.
332  ! Reshape & transpose to be done here! This is temporary!
333  ! Use the read data on number of rows/cols if useful at this point.
334  ! Discuss with Aurore.
335  sll_allocate(mp%local_indices(i)%array(rows, cols), ierr)
336  sll_allocate(local_spline_indices_array(li_total), ierr)
337  read (input_file_id, local_spline_indices)
338  mp%local_indices(i)%array(:, :) = &
339  transpose(reshape(local_spline_indices_array(:), (/cols, rows/)))
340  close (input_file_id)
341  sll_deallocate_array(local_spline_indices_array, ierr)
342  end do
343 
344  print *, 'read_from_file_ctmp2d(): reading local to global index data. '
345 
346  do i = 1, number_patches
347  call sll_s_new_file_id(input_file_id, ierr)
348  if (ierr .ne. 0) then
349  print *, 'ERROR while trying to obtain an unique identifier ', &
350  'for file ', trim(local_to_global_file_list(i)), &
351  '. Called from read_from_file_ctmp2d().'
352  stop
353  end if
354  open (unit=input_file_id, file=trim(local_to_global_file_list(i)), &
355  status="OLD", iostat=io_stat)
356  if (io_stat .ne. 0) then
357  print *, 'ERROR while opening file ', &
358  trim(local_to_global_file_list(i)), &
359  '. Called from read_from_file_ctmp2d().'
360  stop
361  end if
362  read (input_file_id, l2g_dimensions)
363 
364  print *, 'read_from_file_ctmp2d(): reading data from file: ', &
365  trim(local_to_global_file_list(i))
366 
367  ! For now we allocate only a linear array, but a 2D array may be
368  ! more useful, especially since we may need to reshape & transpose.
369  ! Reshape & transpose to be done here! This is temporary!
370  ! Use the read data on number of rows/cols if useful at this point.
371  ! Discuss with Aurore.
372  sll_allocate(mp%local_to_global_ind(i)%array(rows, cols), ierr)
373  sll_allocate(local_to_global_index_array(l2g_total), ierr)
374  read (input_file_id, local_to_global_indices)
375  mp%local_to_global_ind(i)%array(:, :) = &
376  transpose(reshape(local_to_global_index_array(:), (/cols, rows/)))
377  close (input_file_id)
378  sll_deallocate_array(local_to_global_index_array, ierr)
379  end do
380 
381  sll_deallocate_array(local_index_file_list, ierr)
382  sll_deallocate_array(local_to_global_file_list, ierr)
383  sll_deallocate(connectivities, ierr)
384  sll_deallocate(connectivities_reshaped, ierr)
385  end subroutine read_from_file_ctmp2d
