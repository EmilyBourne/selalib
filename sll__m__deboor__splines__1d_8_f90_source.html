475 !*********************************************************************
476 !
477 !
478 ! Discussion:
479 !
480 ! The spline is taken to be continuous from the right.
481 !
482 ! The nontrivial knot interval (T(I),T(I+1)) containing X is
483 ! located with the aid of sll_s_interv. The K B-spline coefficients
484 ! of F relevant for this interval are then obtained from BCOEF,
485 ! or are taken to be zero if not explicitly available, and are
486 ! then differenced JDERIV times to obtain the B-spline
487 ! coefficients of (D**JDERIV)F relevant for that interval.
488 !
489 ! Precisely, with J = JDERIV, we have from X.(12) of the text that:
490 !
491 ! (D**J)F = sum ( BCOEF(.,J)*B(.,K-J,T) )
492 !
493 ! where
494 ! / BCOEF(.), if J == 0
495 ! /
496 ! BCOEF(.,J) = / BCOEF(.,J-1) - BCOEF(.-1,J-1)
497 ! / -----------------------------, if 0 < J
498 ! / (T(.+K-J) - T(.))/(K-J)
499 !
500 ! Then, we use repeatedly the fact that
501 !
502 ! sum ( A(.) * B(.,M,T)(X) ) = sum ( A(.,X) * B(.,M-1,T)(X) )
503 !
504 ! with
505 ! (X - T(.))*A(.) + (T(.+M-1) - X)*A(.-1)
506 ! A(.,X) = ---------------------------------------
507 ! (X - T(.)) + (T(.+M-1) - X)
508 !
509 ! to write (D**J)F(X) eventually as a linear combination of
510 ! B-splines of order 1, and the coefficient for B(I,1,T)(X)
511 ! must then be the desired number (D**J)F(X).
512 ! See Chapter X, (17)-(19) of text.
513 !
514 ! Modified:
515 !
516 ! 14 February 2007
517 !
518 ! Author:
519 !
520 ! Carl DeBoor
521 !
522 ! Reference:
523 !
524 ! Carl DeBoor,
525 ! A Practical Guide to Splines,
526 ! Springer, 2001,
527 ! ISBN: 0387953663.
528 !
529 ! Parameters:
530 !
531 ! Input, real ( kind = 8 ) T(N+K), the knot sequence. T is assumed
532 ! to be nondecreasing.
533 !
534 ! Input, real ( kind = 8 ) BCOEF(N), B-spline coefficient sequence.
535 !
536 ! Input, integer N, the length of BCOEF.
537 !
538 ! Input, integer K, the order of the spline.
539 !
540 ! Input, real ( kind = 8 ) X, the point at which to evaluate.
541 !
542 ! Input, integer JDERIV, the order of the derivative to
543 ! be evaluated. JDERIV is assumed to be zero or positive.
544 !
545 ! Output, real ( kind = 8 ) sll_f_bvalue, the value of the (JDERIV)-th
546 ! derivative of the spline at X.
547 !

581  sll_clear_allocate(dr(1:k), ierr)
582 
583  if (k <= jderiv) return

585 ! Find I so that 1 <= I < N+K and T(I) < T(I+1) and T(I) <= X < T(I+1).
586 !
587 ! If no such I can be found, X lies outside the support of the
588 ! spline F and sll_f_bvalue = 0. The asymmetry in this choice of I makes F
589 ! right continuous.
590 
591  call sll_s_interv(deboor, t, n + k, x, i, mflag)
592 
593  if (mflag /= 0) return
594 !
595 ! If K = 1 (and JDERIV = 0), sll_f_bvalue = BCOEF(I).
596 !
597  if (k <= 1) then
598  sll_f_bvalue = coeff_splines(i)
599  return
600  end if
601 !
602 ! Store the K B-spline coefficients relevant for the knot interval
603 ! ( T(I),T(I+1) ) in AJ(1),...,AJ(K) and compute DL(J) = X - T(I+1-J),
604 ! DR(J) = T(I+J)-X, J=1,...,K-1. Set any of the AJ not obtainable
605 ! from input to zero.
606 !
607 ! Set any T's not obtainable equal to T(1) or to T(N+K) appropriately.
608 !
609  jcmin = 1
610 
611  if (k <= i) then
612 
613  do j = 1, k - 1
614  dl(j) = x - t(i + 1 - j)
615  end do
616 
617  else
618 
619  jcmin = 1 - (i - k)
620  do j = 1, i
621  dl(j) = x - t(i + 1 - j)
622  end do
623  do j = i, k - 1
624  aj(k - j) = 0.0_8
625  dl(j) = dl(i)
626  end do
627 
628  end if
629

631

633

637  end do
638

641  dr(j) = dr(k + n - i)
642  end do

645

647  dr(j) = t(i + j) - x
648  end do
649 
650  end if

653  aj(jc) = coeff_splines(i - k + jc)
654  end do
655 !
656 ! Difference the coefficients JDERIV times.
657 !
658  do j = 1, jderiv
659

661  do jj = 1, k - j
662  aj(jj) = ((aj(jj + 1) - aj(jj))/(dl(ilo) + dr(jj)))*real(k - j, 8)
663  ilo = ilo - 1
664  end do
665 
666  end do

668 ! Compute value at X in (T(I),T(I+1)) of JDERIV-th derivative,
669 ! given its relevant B-spline coefficients in AJ(1),...,AJ(K-JDERIV).
670 !
671  do j = jderiv + 1, k - 1
672  ilo = k - j
673  do jj = 1, k - j
674  aj(jj) = (aj(jj + 1)*dl(ilo) + aj(jj)*dr(jj))/(dl(ilo) + dr(jj))
675  ilo = ilo - 1
676  end do
677  end do
678 
679  sll_f_bvalue = aj(1)

681  return

685 !*************************************************************************
686 !
687 !! sll_s_splint produces the B-spline coefficients BCOEF of an
688 ! interpolating spline.
689 !
690 ! Discussion:
691 !
692 ! The spline is of order K with knots T(1:N+K), and takes on the
693 ! value GTAU(I) at TAU(I), for I = 1 to N.
694 !
695 ! The I-th equation of the linear system
696 !
697 ! A * BCOEF = B
698 !
699 ! for the B-spline coefficients of the interpolant enforces interpolation
700 ! at TAU(1:N).
701 !
702 ! Hence, B(I) = GTAU(I), for all I, and A is a band matrix with 2*K-1
703 ! bands, if it is invertible.
704 !
705 ! The matrix A is generated row by row and stored, diagonal by diagonal,
706 ! in the rows of the array Q, with the main diagonal going
707 ! into row K. See comments in the program.
708 !
709 ! The banded system is then solved by a call to BANFAC, which
710 ! constructs the triangular factorization for A and stores it again in
711 ! Q, followed by a call to BANSLV, which then obtains the solution
712 ! BCOEF by substitution.
713 !
714 ! BANFAC does no pivoting, since the total positivity of the matrix
715 ! A makes this unnecessary.
716 !
717 ! The linear system to be solved is (theoretically) invertible if
718 ! and only if
719 ! T(I) < TAU(I) < TAU(I+K), for all I.
720 ! Violation of this condition is certain to lead to IFLAG = 2.
721 !
722 ! Modified:
723 !
724 ! 14 February 2007
725 !
726 ! Author:
727 !
728 ! Carl DeBoor
729 !
730 ! Reference:
731 !
732 ! Carl DeBoor,
733 ! A Practical Guide to Splines,
734 ! Springer, 2001,
735 ! ISBN: 0387953663,
736 ! LC: QA1.A647.v27.
737 !
738 ! Parameters:
739 !
740 ! Input, real ( kind = 8 ) TAU(N), the data point abscissas.The entries in
741 ! TAU should be strictly increasing.
742 !
743 ! Input, real ( kind = 8 ) GTAU(N), the data ordinates.
744 !
745 ! Input, real ( kind = 8 ) T(N+K), the knot sequence.
746 !
747 ! Input, integer ( kind = 4 ) N, the number of data points.
748 !
749 ! Input, integer ( kind = 4 ) K, the order of the spline.
750 !
751 ! Output, real ( kind = 8 ) Q((2*K-1)*N), the triangular factorization
752 ! of the coefficient matrix of the linear system for the B-coefficients
753 ! of the spline interpolant. The B-coefficients for the interpolant
754 ! of an additional data set can be obtained without going through all
755 ! the calculations in this routine, simply by loading HTAU into BCOEF
756 ! and then executing the call:
757 ! call banslv ( q, 2*k-1, n, k-1, k-1, bcoef )
758 !
759 ! Output, real ( kind = 8 ) BCOEF(N), the B-spline coefficients of
760 ! the interpolant.
761 !
762 ! Output, integer ( kind = 4 ) IFLAG, error flag.
763 ! 1, = success.
764 ! 2, = failure.
765 !
766  subroutine sll_s_splint(deboor, tau, gtau, t, n, k, q, coeff_splines, iflag)
767  implicit none
768

771  sll_real64, dimension(:), intent(in) :: gtau
772  sll_real64, dimension(:), intent(in) :: t

774  sll_int32, intent(in) :: k

776  sll_real64, dimension(:), intent(out) :: coeff_splines
777  sll_int32, intent(out) :: iflag

780  sll_int32 :: jj

789  sll_assert(size(t) == n + k)

796 ! Loop over I to construct the N interpolation equations.
797  do i = 1, n

802  ! Find LEFT in the closed interval (I,I+K-1) such that
803  !
804  ! T(LEFT) <= TAU(I) < T(LEFT+1)
805  !
