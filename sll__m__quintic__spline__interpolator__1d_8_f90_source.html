144 
145  character(len=*), parameter :: this_sub_name = 'initialize'
146  sll_int32 :: ierr
147  sll_int32 :: i
148  sll_real64 :: dx
149 
150  if (bc_min == sll_p_periodic .or. bc_max == sll_p_periodic) then

154  interpolator%n = num_points

159 
160  sll_allocate(interpolator%x(num_points),ierr)
161  dx = (x_max - x_min) / real(num_points - 1,f64)

164  end do
165  interpolator%dx = dx

167  interpolator%ind1 = +1
168  interpolator%indn = +1
169
